```typescript
import { renderHook, waitFor } from '@testing-library/react';
import { useAgentDecisions } from '../useAgentDecisions';

// ── Minimal row factory ───────────────────────────────────────────────────────
function makeRow(overrides: Record<string, unknown> = {}) {
  return {
    id: 'row-id-1',
    decision_id: 'dec-id-1',
    user_id: null,
    symbol: 'EURUSD',
    signal_type: 'BUY',
    confidence: 0.8,
    reasoning: 'test reasoning',
    blockers: [],
    contributions: [],
    signal_mode: 'agent',
    created_at: '2026-05-01T00:00:00Z',
    ...overrides,
  };
}

// ── Mock the supabase module ──────────────────────────────────────────────────
//
// The hook calls:
//   supabase.from('agent_decisions').select('*').order(...).limit(100)[.eq(...)]
//   supabase.channel('...').on('postgres_changes', ...).subscribe(...)
//   supabase.removeChannel(channel)
//
// We need a chainable mock for the query builder and a stub realtime channel.

let mockQueryResult: { data: unknown[] | null; error: { message: string } | null } = {
  data: [],
  error: null,
};
let eqSpy: ReturnType<typeof vi.fn>;

vi.mock('../../lib/supabase', () => {
  eqSpy = vi.fn().mockImplementation(() => Promise.resolve(mockQueryResult));

  const queryBuilder = {
    select: vi.fn().mockReturnThis(),
    order: vi.fn().mockReturnThis(),
    limit: vi.fn().mockImplementation(() => ({
      ...queryBuilder,
      // Resolve immediately when awaited (no .eq chained)
      then: (resolve: (v: typeof mockQueryResult) => void) =>
        Promise.resolve(mockQueryResult).then(resolve),
      eq: eqSpy,
    })),
  };

  const channelStub = {
    on: vi.fn().mockReturnThis(),
    subscribe: vi.fn().mockReturnThis(),
  };

  return {
    supabase: {
      from: vi.fn().mockReturnValue(queryBuilder),
      channel: vi.fn().mockReturnValue(channelStub),
      removeChannel: vi.fn(),
    },
  };
});

// ── Tests ─────────────────────────────────────────────────────────────────────

describe('useAgentDecisions — RLS-03 empty-feed contract', () => {
  beforeEach(() => {
    mockQueryResult = { data: [], error: null };
  });

  it('Test 1: returns shared (user_id=null) rows when own rows are absent', async () => {
    mockQueryResult = {
      data: [makeRow({ id: 'r1', user_id: null }), makeRow({ id: 'r2', user_id: null })],
      error: null,
    };

    const { result } = renderHook(() => useAgentDecisions());
    await waitFor(() => expect(result.current.loading).toBe(false));

    expect(result.current.decisions).toHaveLength(2);
    expect(result.current.error).toBeNull();
    // Both rows have user_id: null — they are shared decisions
    expect(result.current.decisions[0].user_id).toBeNull();
    expect(result.current.decisions[1].user_id).toBeNull();
  });

  it('Test 2: returns own rows (user_id set) when they exist', async () => {
    mockQueryResult = {
      data: [makeRow({ id: 'r1', user_id: 'user-abc' })],
      error: null,
    };

    const { result } = renderHook(() => useAgentDecisions());
    await waitFor(() => expect(result.current.loading).toBe(false));

    expect(result.current.decisions).toHaveLength(1);
    expect(result.current.decisions[0].user_id).toBe('user-abc');
    expect(result.current.error).toBeNull();
  });

  it('Test 3: applies symbolFilter as .eq() on the query', async () => {
    mockQueryResult = { data: [], error: null };

    const { result } = renderHook(() => useAgentDecisions({ symbolFilter: 'EURUSD' }));
    await waitFor(() => expect(result.current.loading).toBe(false));

    // eqSpy is called by .eq('symbol', 'EURUSD') in the query builder
    expect(eqSpy).toHaveBeenCalledWith('symbol', 'EURUSD');
  });

  it('Test 4: surfaces Supabase query errors as hook.error', async () => {
    mockQueryResult = { data: null, error: { message: 'permission denied' } };

    const { result } = renderHook(() => useAgentDecisions());
    await waitFor(() => expect(result.current.loading).toBe(false));

    expect(result.current.error).toBe('permission denied');
    expect(result.current.decisions).toEqual([]);
  });

  it('Test 5: does not filter away null user_id rows (regression guard for RLS-03)', async () => {
    // If a future change adds client-side filtering like:
    //   .filter(row => row.user_id === user?.id)
    // this test will fail because user_id: null would be filtered out.
    mockQueryResult = {
      data: [makeRow({ id: 'r1', user_id: null, symbol: 'GBPUSD', signal_type: 'SELL' })],
      error: null,
    };

    const { result } = renderHook(() => useAgentDecisions());
    await waitFor(() => expect(result.current.loading).toBe(false));

    expect(result.current.decisions).toHaveLength(1);
    expect(result.current.decisions[0].user_id).toBeNull();
    expect(result.current.decisions[0].symbol).toBe('GBPUSD');
    // TODO Phase 5 (CI-03): replace this mock test with a real Supabase local stack
    // policy fixture test that verifies the actual SQL SELECT policy behavior.
  });
});
```
